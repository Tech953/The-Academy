5️⃣ core/terminal.js
export class Terminal {
  constructor(os) {
    this.os = os;
    this.output = document.getElementById("output");
    this.input = document.getElementById("command-input");
  }

  print(text) {
    this.output.textContent += text + "\n";
  }

  listen() {
    this.input.addEventListener("keydown", (e) => {
      if (e.key === "Enter") {
        const cmd = this.input.value.trim();
        this.input.value = "";
        this.handle(cmd);
      }
    });
  }

  handle(command) {
    this.print(`> ${command}`);

    const [cmd, ...args] = command.split(" ");

    if (cmd === "help") {
      this.print("Commands: help, ls, open, run, clear");
    } 
    else if (cmd === "clear") {
      this.output.textContent = "";
    }
    else {
      this.os.apps.run(cmd, args, this);
    }
  }
}

export class Terminal {
  constructor(os) {
    this.os = os;
    this.output = document.getElementById("output");
    this.input = document.getElementById("command-input");
  }

  print(text) {
    this.output.textContent += text + "\n";
    this.output.scrollTop = this.output.scrollHeight;
  }

  listen() {
    this.input.addEventListener("keydown", (e) => {
      if (e.key === "Enter") {
        const cmd = this.input.value.trim();
        this.input.value = "";
        this.handle(cmd);
      }
    });
  }

  handle(command) {
    if (!command) return;
    this.print(`> ${command}`);

    const [cmd, ...args] = command.split(" ");

    switch (cmd) {
      case "help":
        this.print(this.os.apps.help());
        break;
      case "clear":
        this.output.textContent = "";
        break;
      default:
        this.os.apps.run(cmd, args, this);
    }
  }
}

3️⃣ ACCESSIBILITY ENGINE
core/accessibility.js
export class AccessibilityManager {
  constructor(os) {
    this.os = os;
    this.profile = "default";
    this.settings = {};
  }

  async loadProfiles() {
    const res = await fetch("data/accessibility_profiles.json");
    this.profiles = await res.json();
    this.applyProfile(this.profile);
  }

  applyProfile(name) {
    if (!this.profiles[name]) return;
    this.profile = name;
    this.settings = this.profiles[name];

    document.documentElement.style.fontSize = this.settings.fontSize + "px";
    document.body.classList.toggle("high-contrast", this.settings.highContrast);
    document.body.classList.toggle("reduced-motion", this.settings.reducedMotion);
    document.body.classList.toggle("dyslexia-font", this.settings.dyslexiaFont);

    this.os.terminal.print(
      `[ACCESSIBILITY] Profile applied: ${this.settings.name}`
    );
  }

  listProfiles() {
    return Object.keys(this.profiles);
  }
}

export class I18n {
  constructor(os) {
    this.os = os;
    this.language = "en";
    this.strings = {};
  }

  async loadLanguages() {
    const res = await fetch("data/languages.json");
    this.languages = await res.json();
    this.setLanguage(this.language);
  }

  setLanguage(code) {
    if (!this.languages[code]) return;
    this.language = code;
    this.strings = this.languages[code].strings;
    this.os.terminal.print(`[LANG] Language set: ${this.languages[code].name}`);
  }

  t(key) {
    return this.strings[key] || key;
  }

  listLanguages() {
    return Object.keys(this.languages);
  }
}

  run(args, terminal, os) {
    if (!args.length) {
      terminal.print("Profiles:");
      os.accessibility.listProfiles().forEach(p => terminal.print(`- ${p}`));
      terminal.print("Usage: accessibility <profile>");

  }
